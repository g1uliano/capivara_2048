```dart
import 'dart:async';
import 'package:flutter/material.dart';
import 'package:flutter_animate/flutter_animate.dart';
import '../../data/animals_data.dart';
import '../widgets/game_background.dart';
import '../widgets/game_title_image.dart';
import 'home_screen.dart';

class SplashScreen extends StatefulWidget {
  const SplashScreen({super.key});

  @override
  State<SplashScreen> createState() => _SplashScreenState();
}

class _SplashScreenState extends State<SplashScreen> {
  bool _initDone = false;
  bool _timerDone = false;

  @override
  void initState() {
    super.initState();
    Timer(const Duration(milliseconds: 1500), () {
      _timerDone = true;
      _tryNavigate();
    });
    _init();
  }

  Future<void> _init() async {
    // precacheImage requires a BuildContext — defer to didChangeDependencies
  }

  @override
  void didChangeDependencies() {
    super.didChangeDependencies();
    if (_initDone) return;
    _precacheAll();
  }

  Future<void> _precacheAll() async {
    final futures = <Future<void>>[];
    for (final animal in animals) {
      futures.add(precacheImage(
        ResizeImage(AssetImage(animal.tilePngPath), width: 144, height: 144),
        context,
      ));
      futures.add(precacheImage(
        ResizeImage(AssetImage(animal.hostPngPath), width: 304, height: 304),
        context,
      ));
    }
    futures.add(precacheImage(const AssetImage('assets/images/fundo.png'), context));
    futures.add(precacheImage(const AssetImage('assets/icons/inventory/bomb_2.png'), context));
    futures.add(precacheImage(const AssetImage('assets/icons/inventory/bomb_3.png'), context));
    futures.add(precacheImage(const AssetImage('assets/icons/inventory/undo_1.png'), context));
    futures.add(precacheImage(const AssetImage('assets/icons/inventory/undo_3.png'), context));
    futures.add(precacheImage(const AssetImage('assets/images/title/title_orange.png'), context));
    await Future.wait(futures);
    _initDone = true;
    _tryNavigate();
  }

  void _tryNavigate() {
    if (!_initDone || !_timerDone) return;
    if (!mounted) return;
    Navigator.of(context).pushReplacement(
      PageRouteBuilder(
        pageBuilder: (_, __, ___) => const HomeScreen(),
        transitionsBuilder: (_, animation, __, child) =>
            FadeTransition(opacity: animation, child: child),
        transitionDuration: const Duration(milliseconds: 400),
      ),
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: GameBackground(
        child: Center(
          child: const GameTitleImage()
              .animate()
              .fadeIn(duration: 600.ms)
              .scale(begin: const Offset(0.95, 0.95), end: const Offset(1, 1), duration: 600.ms),
        ),
      ),
    );
  }
}
```

```dart
class GameState {
  final List<List<Tile?>> board;
  final int score;
  final int highScore;
  final bool isGameOver;
  final bool hasWon;
  final int maxLevel;
  final int elapsedMs;
  final bool isPaused;
  final List<GameState> undoStack;
  final BombMode? bombMode;
  final List<(int, int)> selectedBombTiles;
  final bool isAwaitingGameOverResolution;   // ← new

  static const _bombSentinel = Object();

  GameState({
    required this.board,
    required this.score,
    required this.highScore,
    required this.isGameOver,
    required this.hasWon,
    this.maxLevel = 0,
    this.elapsedMs = 0,
    this.isPaused = false,
    List<GameState>? undoStack,
    this.bombMode,
    this.selectedBombTiles = const [],
    this.isAwaitingGameOverResolution = false,  // ← new
  }) : undoStack = List.unmodifiable(undoStack ?? const []);

  GameState copyWith({
    List<List<Tile?>>? board,
    int? score,
    int? highScore,
    bool? isGameOver,
    bool? hasWon,
    int? maxLevel,
    int? elapsedMs,
    bool? isPaused,
    List<GameState>? undoStack,
    Object? bombMode = _bombSentinel,
    List<(int, int)>? selectedBombTiles,
    bool? isAwaitingGameOverResolution,  // ← new
  }) {
    return GameState(
      board: board ?? this.board,
      score: score ?? this.score,
      highScore: highScore ?? this.highScore,
      isGameOver: isGameOver ?? this.isGameOver,
      hasWon: hasWon ?? this.hasWon,
      maxLevel: maxLevel ?? this.maxLevel,
      elapsedMs: elapsedMs ?? this.elapsedMs,
      isPaused: isPaused ?? this.isPaused,
      undoStack: undoStack ?? this.undoStack,
      bombMode: bombMode == _bombSentinel ? this.bombMode : bombMode as BombMode?,
      selectedBombTiles: selectedBombTiles ?? this.selectedBombTiles,
      isAwaitingGameOverResolution:                          // ← new
          isAwaitingGameOverResolution ?? this.isAwaitingGameOverResolution,
    );
  }
}
```
