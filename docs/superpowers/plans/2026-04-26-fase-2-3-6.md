```dart
class GameScreen extends ConsumerStatefulWidget {
  const GameScreen({super.key});

  @override
  ConsumerState<GameScreen> createState() => _GameScreenState();
}

class _GameScreenState extends ConsumerState<GameScreen> {
  final GlobalKey _headerKey = GlobalKey();
  double _pauseTop = 80;

  @override
  void initState() {
    super.initState();
    WidgetsBinding.instance.addPostFrameCallback((_) => _updatePausePosition());
  }

  void _updatePausePosition() {
    final renderBox =
        _headerKey.currentContext?.findRenderObject() as RenderBox?;
    if (renderBox == null) return;
    final headerBottom =
        renderBox.localToGlobal(Offset.zero).dy + renderBox.size.height;
    if (mounted) {
      setState(() => _pauseTop = headerBottom + 12);
    }
  }

  @override
  Widget build(BuildContext context) {
    final state = ref.watch(gameProvider);
    final isGameOver = state.isGameOver;
    final isPaused = state.isPaused;
    final bombMode = state.bombMode;
    final notifier = ref.read(gameProvider.notifier);
    final hostAnimal =
        state.maxLevel > 0 ? animalForLevel(state.maxLevel) : null;

    ref.listen<GameState>(gameProvider, (prev, next) {
      if (prev != null && !prev.isGameOver && next.isGameOver && !next.hasWon) {
        ref.read(livesProvider.notifier).consume();
      }
    });

    return Scaffold(
      body: GameBackground(
        animal: hostAnimal,
        child: SafeArea(
          child: Stack(
            children: [
              // Fixed column — never adds or removes children
              Column(
                children: [
                  Padding(
                    key: _headerKey,
                    padding: const EdgeInsets.symmetric(
                        horizontal: 16, vertical: 8),
                    child: Row(
                      crossAxisAlignment: CrossAxisAlignment.start,
                      children: [
                        const HostBanner(),
                        const Spacer(),
                        const StatusPanel(),
                      ],
                    ),
                  ),
                  const Spacer(),
                  GestureDetector(
                    behavior: HitTestBehavior.opaque,
                    onPanEnd: (details) {
                      if (isPaused || isGameOver || bombMode != null) return;
                      final v = details.velocity.pixelsPerSecond;
                      const threshold = 100.0;
                      if (v.dx.abs() > v.dy.abs()) {
                        if (v.dx > threshold) {
                          notifier.onSwipe(Direction.right);
                        } else if (v.dx < -threshold) {
                          notifier.onSwipe(Direction.left);
                        }
                      } else {
                        if (v.dy > threshold) {
                          notifier.onSwipe(Direction.down);
                        } else if (v.dy < -threshold) {
                          notifier.onSwipe(Direction.up);
                        }
                      }
                    },
                    child: const RepaintBoundary(child: BoardWidget()),
                  ),
                  const Spacer(),
                  const InventoryBar(),
                  const Padding(
                    padding:
                        EdgeInsets.symmetric(horizontal: 16, vertical: 8),
                    child: LivesIndicator(),
                  ),
                ],
              ),
              // Overlays — Positioned.fill so board never shifts
              if (isPaused) const Positioned.fill(child: PauseOverlay()),
              if (isGameOver) const Positioned.fill(child: GameOverModal()),
              // Pause button — hidden during game over
              if (!isGameOver)
                Positioned(
                  top: _pauseTop,
                  right: 8,
                  child: IconButton(
                    icon: Icon(
                      isPaused
                          ? Icons.play_arrow_rounded
                          : Icons.pause_rounded,
                      color: Colors.white,
                    ),
                    iconSize: 32,
                    onPressed: isPaused ? notifier.resume : notifier.pause,
                  ),
                ),
            ],
          ),
        ),
      ),
    );
  }
}
```

```dart
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:shared_preferences/shared_preferences.dart';

class ReduceEffectsNotifier extends StateNotifier<bool> {
  ReduceEffectsNotifier() : super(false);

  Future<void> load() async {
    final prefs = await SharedPreferences.getInstance();
    state = prefs.getBool('reduce_effects') ?? false;
  }

  Future<void> toggle() async {
    state = !state;
    final prefs = await SharedPreferences.getInstance();
    await prefs.setBool('reduce_effects', state);
  }
}

final reduceEffectsProvider =
    StateNotifierProvider<ReduceEffectsNotifier, bool>((ref) {
  return ReduceEffectsNotifier();
});
```

```dart
import 'package:hive/hive.dart';
import 'inventory.dart';

class InventoryHiveAdapter extends TypeAdapter<Inventory> {
  @override
  final int typeId = 2;

  @override
  Inventory read(BinaryReader reader) {
    final numOfFields = reader.readByte();
    final fields = <int, dynamic>{
      for (int i = 0; i < numOfFields; i++) reader.readByte(): reader.read(),
    };
    return Inventory(
      bomb2: fields[0] as int,
      bomb3: fields[1] as int,
      undo1: fields[2] as int,
      undo3: fields[3] as int,
    );
  }

  @override
  void write(BinaryWriter writer, Inventory obj) {
    writer
      ..writeByte(4)
      ..writeByte(0)
      ..write(obj.bomb2)
      ..writeByte(1)
      ..write(obj.bomb3)
      ..writeByte(2)
      ..write(obj.undo1)
      ..writeByte(3)
      ..write(obj.undo3);
  }

  @override
  int get hashCode => typeId.hashCode;

  @override
  bool operator ==(Object other) =>
      identical(this, other) ||
      other is InventoryHiveAdapter &&
          runtimeType == other.runtimeType &&
          typeId == other.typeId;
}
```

```dart
class InventoryNotifier extends StateNotifier<Inventory> {
  final InventoryRepository _repo;

  InventoryNotifier(this._repo) : super(Inventory.empty());

  Future<void> load() async {
    state = await _repo.load();
  }

  Future<void> consume(ItemType type) async {
    final current = _countOf(type);
    if (current <= 0) return;
    state = _setCount(state, type, current - 1);
    await _repo.save(state);
  }

  Future<void> add(ItemType type, int amount) async {
    state = _setCount(state, type, _countOf(type) + amount);
    await _repo.save(state);
  }

  int count(ItemType type) => _countOf(type);

  Future<void> addDebugItems() async {
    assert(kDebugMode);
    for (final type in ItemType.values) {
      await add(type, 5);
    }
  }

  int _countOf(ItemType type) {
    switch (type) {
      case ItemType.bomb2: return state.bomb2;
      case ItemType.bomb3: return state.bomb3;
      case ItemType.undo1: return state.undo1;
      case ItemType.undo3: return state.undo3;
    }
  }

  Inventory _setCount(Inventory inv, ItemType type, int value) {
    switch (type) {
      case ItemType.bomb2: return inv.copyWith(bomb2: value);
      case ItemType.bomb3: return inv.copyWith(bomb3: value);
      case ItemType.undo1: return inv.copyWith(undo1: value);
      case ItemType.undo3: return inv.copyWith(undo3: value);
    }
  }
}
```

```dart
    if (state.isGameOver) return state;

    final rotated = _rotateBoard(state.board, dir);
    int totalGained = 0;
    bool anyChanged = false;

    final newBoard = rotated.map((row) {
      final result = _compactAndMerge(row);
      if (result.changed) anyChanged = true;
      totalGained += result.gained;
      return result.row;
    }).toList();

    if (!anyChanged) {
      final isOver = _checkGameOver(state.board);
      return state.copyWith(isGameOver: isOver);
    }

    final rawUnrotated = _unrotateBoard(newBoard, dir);
    final size = GameConstants.boardSize;
    final unrotated = List.generate(size, (r) =>
      List.generate(size, (c) {
        final t = rawUnrotated[r][c];
        return t != null ? t.copyWith(row: r, col: c) : null;
      })
    );
    final newScore = state.score + totalGained;
    final newHighScore = max(state.highScore, newScore);
    final hasWon = _checkWin(unrotated);

    int newMaxLevel = state.maxLevel;
    for (final row in unrotated) {
      for (final tile in row.whereType<Tile>()) {
        if (tile.level > newMaxLevel) newMaxLevel = tile.level;
      }
    }

    // Push current state onto undoStack (max 3)
    final newUndoStack = List<GameState>.unmodifiable([
      state,
      ...state.undoStack.take(2),
    ]);

    var next = state.copyWith(
      board: unrotated,
      score: newScore,
      highScore: newHighScore,
      hasWon: hasWon,
      maxLevel: newMaxLevel,
      undoStack: newUndoStack,
    );

    final isGameOver = _checkGameOver(next.board);
    return next.copyWith(isGameOver: isGameOver);
```

```dart
  final List<(int, int)> _bombSelection = [];

  void enterBombMode(BombMode mode) {
    if (state.isGameOver || state.isPaused) return;
    _bombSelection.clear();
    state = state.copyWith(bombMode: mode);
  }

  void selectBombTile(int r, int c) {
    final mode = state.bombMode;
    if (mode == null) return;
    final tile = state.board[r][c];
    if (tile == null) return; // empty — caller shows SnackBar

    final pos = (r, c);
    final alreadySelected = _bombSelection.contains(pos);

    if (alreadySelected) {
      _bombSelection.remove(pos);
      state = state.copyWith(); // trigger rebuild
      return;
    }

    if (mode == BombMode.bomb2) {
      if (_bombSelection.isEmpty) {
        _bombSelection.add(pos);
      } else {
        final first = _bombSelection[0];
        final isAdjacent =
            (r - first.$1).abs() + (c - first.$2).abs() == 1;
        if (isAdjacent) {
          _bombSelection.add(pos);
        } else {
          // Replace first selection with new tile
          _bombSelection
            ..clear()
            ..add(pos);
        }
      }
    } else {
      // bomb3 — add up to 3
      if (_bombSelection.length < 3) {
        _bombSelection.add(pos);
      }
    }
    state = state.copyWith(); // trigger rebuild
  }

  void confirmBomb() {
    final mode = state.bombMode;
    if (mode == null) return;
    final required = mode == BombMode.bomb2 ? 2 : 3;
    if (_bombSelection.length < required) return;
    final newState = _engine.removeTiles(state, List.from(_bombSelection));
    _bombSelection.clear();
    state = newState.copyWith(bombMode: null);
  }

  void cancelBomb() {
    _bombSelection.clear();
    state = state.copyWith(bombMode: null);
  }
```
