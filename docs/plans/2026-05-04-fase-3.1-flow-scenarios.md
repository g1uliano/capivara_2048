## Contexto técnico (ler antes de implementar)

### Arquitetura relevante de GameState

```dart
// Campos que os cenários manipulam via debugSetState:
GameState {
  board: List<List<Tile?>>,   // 4×4
  score: int,
  isGameOver: bool,
  hasWon: bool,
  isPaused: bool,
  maxLevel: int,
  pendingMilestone: int?,     // != null → VictoryChoiceDialog aparece
  isAwaitingGameOverResolution: bool,  // true → GameOverItemOverlay ou GameOverNoItemsOverlay
  isContinuingWithItem: bool,
}
```

### Qual overlay aparece na GameScreen

| Condição | Widget mostrado |
|---|---|
| `isPaused` | `PauseOverlay` (botões: 'Continuar', 'Reiniciar', 'Menu') |
| `isAwaitingGameOverResolution && hasAnyItem` | `GameOverItemOverlay` |
| `isAwaitingGameOverResolution && !hasAnyItem` | `GameOverNoItemsOverlay` (texto: 'Você não possui mais itens!') |
| `isGameOver && !isAwaitingGameOverResolution && !isContinuingWithItem` | `GameOverModal` ('Game Over!') |
| `hasWon && !isGameOver` | `GameOverModal` ('Capivara Lendária! 🎉') |
| `pendingMilestone != null && !hasWon` | `VictoryChoiceDialog` |

### Milestones / níveis de vitória

| Milestone | Nível | Texto em VictoryChoiceDialog |
|---|---|---|
| 11 | 2048 | 'Você chegou ao 2048!' |
| 12 | 4096 | 'Você chegou ao 4096!' |
| 13 | 8192 | 'Você chegou ao 8192!' |

### Tiles: `Tile(id: String, level: int)`

Level 10 = nível pré-vitória (precursor do 2048). Para forçar vitória: dois `Tile(level:10)` adjacentes → swipe → merge em `Tile(level:11)` → `pendingMilestone = 11`.

/// Lê o GameState atual do container.
GameState readGame(GameTestHarness harness) =>
    harness.container.read(gameProvider);

/// Define um estado de jogo com dois tiles de [level-1] prontos para fundir.
/// Após swipeBoard(SwipeDirection.left), o merge produz tile de [level],
/// disparando pendingMilestone = [level] (se level >= 11).
void setupNearWin(GameTestHarness harness, {int level = 11}) {
  final board = List.generate(4, (_) => List<Tile?>.filled(4, null));
  board[0][0] = Tile(id: 'near_win_a', level: level - 1);
  board[0][1] = Tile(id: 'near_win_b', level: level - 1);
  harness.container.read(gameProvider.notifier).debugSetState(
    GameState(
      board: board,
      score: 100,
      highScore: 0,
      isGameOver: false,
      hasWon: false,
      maxLevel: level - 1,
    ),
  );
}

/// Define estado de game over aguardando resolução (mostra GameOverItemOverlay
/// ou GameOverNoItemsOverlay dependendo do inventário).
void setupAwaitingResolution(GameTestHarness harness) {
  harness.container.read(gameProvider.notifier).debugSetState(
    GameState(
      board: List.generate(4, (_) => List<Tile?>.filled(4, null)),
      score: 100,
      highScore: 0,
      isGameOver: true,
      hasWon: false,
      isAwaitingGameOverResolution: true,
    ),
  );
}

/// Define estado de game over final (mostra GameOverModal com 'Game Over!').
void setupGameOverModal(GameTestHarness harness) {
  harness.container.read(gameProvider.notifier).debugSetState(
    GameState(
      board: List.generate(4, (_) => List<Tile?>.filled(4, null)),

      highScore: 0,
      isGameOver: true,
      hasWon: false,
      isAwaitingGameOverResolution: false,

/// Regras do harness que se repetem nos dois cenários.
Future<void> _bootToGame(tester, harness) async {
  await tester.pumpWidget(await harness.boot());
  await tester.pumpAndSettle(const Duration(seconds: 5));
  await tester.gotoGame(harness);

// ─── flow.continue_after_pause ───────────────────────────────────────────────

final continueAfterPauseScenario = E2EScenario(
  id: 'flow.continue_after_pause',
  title: 'pausar → tap "Continuar" → jogo retoma (sem PauseOverlay)',
  tags: {ScenarioTag.critical},
  run: (tester, harness) async {
    await _bootToGame(tester, harness);

    // Pausa via notifier (tap no botão de pausa pode ser flaky por tamanho;
    // chamar direto é mais confiável no Tier 1 headless).
    harness.container.read(gameProvider.notifier).pause();
    await tester.pump();

    expect(find.text('Pausado'), findsOneWidget);
    expect(tester.readGame(harness).isPaused, isTrue);

    await tester.tap(find.text('Continuar'));
    await tester.pumpAndSettle();

    expect(find.text('Pausado'), findsNothing);
    expect(tester.readGame(harness).isPaused, isFalse);
    expect(find.byType(GameScreen), findsOneWidget);
  },
);

// ─── flow.continue_after_back_button ─────────────────────────────────────────

final continueAfterBackButtonScenario = E2EScenario(
  id: 'flow.continue_after_back_button',
  title: 'pause → Menu → home → "Continuar Jogo" → SEM PauseOverlay (regressão v1.2.9)',
  tags: {ScenarioTag.critical},
  run: (tester, harness) async {
    await _bootToGame(tester, harness);

    // Pausa e volta ao menu via overlay.
    harness.container.read(gameProvider.notifier).pause();
    await tester.pump();
    expect(find.text('Pausado'), findsOneWidget);

    await tester.tap(find.text('Menu'));
    await tester.pumpAndSettle();

    // Deve estar na Home com 'Continuar Jogo' visível.
    expect(find.text('Continuar Jogo'), findsOneWidget);

    // Toca em 'Continuar Jogo'.
    await tester.tap(find.text('Continuar Jogo'));
    await tester.pumpAndSettle();

    // Regressão v1.2.9: PauseOverlay NÃO deve aparecer.
    expect(find.text('Pausado'), findsNothing,
        reason: 'regressão v1.2.9 — _continueGame deve chamar resume()');
    expect(find.byType(GameScreen), findsOneWidget);
  },
);

final useUndoDuringGameScenario = E2EScenario(
  id: 'flow.use_undo_during_game',
  title: 'undo1: fazer 2 movimentos → usar Desfazer 1 → estado reverte',
  tags: {ScenarioTag.critical},
  run: (tester, harness) async {
    await _bootToGame(tester, harness);

    // Garantir undo1 disponível.
    harness.container
        .read(inventoryProvider.notifier)
        .debugSetState(const Inventory(bomb2: 0, bomb3: 0, undo1: 2, undo3: 0));

    // Fazer 2 movimentos para popular o undoStack.
    await tester.swipeBoard(SwipeDirection.left);
    await tester.swipeBoard(SwipeDirection.right);
    final scoreAfterMoves = tester.readGame(harness).score;
    final inventoryBefore =
        harness.container.read(inventoryProvider).undo1;

    // Usar Desfazer 1 (tap no botão).
    await tester.tap(find.byKey(const Key('inventory_undo1')));
    await tester.pumpAndSettle();

    // Inventário deve ter decrementado.
    final inventoryAfter = harness.container.read(inventoryProvider).undo1;
    expect(inventoryAfter, lessThan(inventoryBefore),
        reason: 'undo1 deve ser consumido ao usar');
    // Score não aumenta após undo.
    expect(tester.readGame(harness).score,
        lessThanOrEqualTo(scoreAfterMoves));
